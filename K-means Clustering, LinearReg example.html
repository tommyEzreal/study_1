In [19]:
from sklearn.cluster import KMeans

inertia_list=[]

for i in range(1,11):
    #사이킷런 KMeans 패키지 
    kmeans = KMeans(n_clusters=i, init = 'k-means++', random_state=0)
    kmeans.fit(X) # 학습시킨 X값(클러스터링한 X값)
    inertia_list.append(kmeans.inertia_) ㅠ
    # 각 데이터들과 클러스터의 중심까지의 거리제곱합
    
plt.plot(range(1,11), inertia_list)
plt.title('Elbow method')
plt.xlabel('n_labels')
plt.ylabel('inertia')
plt.show()
